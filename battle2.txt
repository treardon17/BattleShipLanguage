player Tyler
computer Putin
Tyler = newPlayer("Tyler")
Putin = newComputer("Putin")
Tyler->randomlyAddBoats(9)
Putin->randomlyAddBoats(9)
loop:
  Tyler->attack("")->Putin
  if Putin->findsBoat():
    begintry:
      1.try: Putin->trySink("up",1)->Tyler
      2.try: Putin->trySink("down",1)->Tyler
      3.try: Putin->trySink("left",1)->Tyler
      4.try: Putin->trySink("right",1)->Tyler
    endtry.
    Putin->show()
  endif.
  else:
    Putin->seek(3)->Tyler
  endelse.
endloop.
